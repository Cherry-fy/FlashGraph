00219     void flush_requests() {
00220         if (task_buf.empty())
00221             return;
00222         // We can't issue any I/O requests to SAFS.
00223         if (io->num_pending_ios() >= io->get_max_num_pending_ios())

00227         // as possible. Each time we issue a request to read at least one
00228         // page. We'll merge requests if the pages touched by user tasks
00229         // in the input array are adjacent to each other.

00233         // The offset of the first page accessed by the I/O request.
00234         const TaskType task = task_buf.front();
00235         task_buf.pop_front();
00236         off_t first_page_off = ROUND_PAGE(task.get_idx() * sizeof(ValueType));
00237         // The offset of the last page accessed by the I/O request.
00238         // The page is excluded by the I/O request.
00239         off_t last_page_off = ROUNDUP_PAGE((task.get_idx()
00240                     + task.get_num_entries()) * sizeof(ValueType));

00242             = (KV_compute<ValueType, TaskType> *) alloc.alloc();

00246                 && num_reqs < avail_io_slots - 1) {
00247             const TaskType task = task_buf.front();
00248             task_buf.pop_front();
00249             off_t page_off = ROUND_PAGE(task.get_idx() * sizeof(ValueType));
00250             off_t end_page_off = ROUNDUP_PAGE((task.get_idx()
00251                     + task.get_num_entries()) * sizeof(ValueType));

00253             if ((page_off >= first_page_off && page_off <= last_page_off)
00254                     || (end_page_off >= first_page_off && end_page_off <= last_page_off)) {

00257                 last_page_off = std::max(end_page_off, last_page_off);

00262             // by the current I/O request.

00269             compute->add_task(task);
