00626     void init(const io_request &req) {
00627         data_loc_t loc(req.get_file_id(), req.get_offset());
00628         assert(!data_inline);
00629         if (req.payload_type == USER_COMPUTE
00630                 || this->payload_type == USER_COMPUTE) {
00631             assert(req.payload_type == USER_COMPUTE
00632                     && this->payload_type == USER_COMPUTE);
00633             // TODO
00634         }
00635         else if (!req.is_extended_req()) {
00636             this->init(req.get_buf(), loc, req.get_size(),
00637                     req.get_access_method(), req.get_io(), req.get_node_id());
00638         }
00639         // Both requests have extensions.
00640         else if (this->is_extended_req()) {
00641             this->init(NULL, loc, 0, req.get_access_method(),
00642                     req.get_io(), req.get_node_id());
00643             this->get_extension()->init(*req.get_extension());
00644         }
00645         else {
00646             // The last case is that this request doesn't have extension,
00647             // but the given request has extension. This request can't keep
00648             // all information in the given request.
00649             assert(!this->is_extended_req() && req.is_extended_req());
00650         }
00651         copy_flags(req);
00652         this->user_data_addr = req.user_data_addr;

00970     /*
00971      * We need to serialize an io request to a buffer so it can be sent to
00972      * another thread.
00973      * @accept_inline: indicates whether the IO request can inline data
00974      * to the buffer.
00975      */
00976     int serialize(char *buf, int size, bool accept_inline) {
00977         int serialized_size;
00978         if (is_data_inline()) {
00979             assert(accept_inline);
00980             serialized_size = get_serialized_size();
00981             assert(serialized_size <= size);
00982             memcpy(buf, this, serialized_size);
00983         }
00984         else if (payload_type == EXT_REQ) {
00985             // We never serialize the io request extension to the message.
00986             serialized_size = sizeof(io_request);
00987             assert(serialized_size <= size);
00988             memcpy(buf, this, sizeof(*this));
00989         }
00990         else if (payload_type == BASIC_REQ) {
00991             // We only serialize the data buffer to the message for write
00992             // requests. The size of the data buffer has to be small.
00993             if (get_size() > MAX_INLINE_SIZE || access_method == READ
00994                     || !accept_inline) {
00995                 serialized_size = sizeof(io_request);
00996                 assert(serialized_size <= size);
00997                 memcpy(buf, this, sizeof(*this));
00998             }
00999             else {
01000                 serialized_size = sizeof(io_request) - sizeof(this->payload)
01001                     + get_size();
01002                 assert(serialized_size <= size);
01003                 memcpy(buf, this, sizeof(*this));
01004                 io_request *p = (io_request *) buf;
01005                 p->data_inline = 1;
01006                 memcpy(p->payload.buf, (char *) this->payload.buf_addr,
01007                         this->get_size());
01008             }
01009         }
01010         else {
01011             // The user compute object is always serialized to the message.
01012             serialized_size = sizeof(io_request);
01013             assert(serialized_size <= size && sizeof(io_request)
01014                     <= (unsigned) size);
01015             memcpy(buf, this, serialized_size);
01016         }
01017         return serialized_size;
01018     }

01086 /*
01087  * Perform the same function to the requests with the same IO instance.
01088  * It turns out it's a common function when delivering requests to
01089  * the upper layer.
01090  */
01091 template<class req_type, class get_io_func, class process_req_func>
01092 void process_reqs_on_io(req_type reqs[], int num,
01093         get_io_func func, process_req_func proc_func)
01094 {
01095     comp_req_io<req_type, get_io_func> req_io_comparator(func);
01096 
01097     std::sort(reqs, reqs + num, req_io_comparator);
01098     io_interface *prev = func(reqs[0]);
01099     int begin_idx = 0;
01100     for (int end_idx = 1; end_idx < num; end_idx++) {
01101         if (func(reqs[end_idx]) != prev) {
01102             proc_func(prev, reqs + begin_idx, end_idx - begin_idx);
01103             begin_idx = end_idx;
01104             prev = func(reqs[end_idx]);
01105         }
01106     }
01107     assert(begin_idx < num);
01108     proc_func(prev, reqs + begin_idx, num - begin_idx);
01109 }
