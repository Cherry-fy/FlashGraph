00323     void max_val_locs(size_t num, std::vector<std::pair<T, off_t> > &pairs) const {
00324         typedef std::pair<T, off_t> val_loc_t;
00325         struct comp_val {
00326             bool operator()(const val_loc_t &v1, const val_loc_t &v2) {
00327                 return v1.first > v2.first;

00330         std::priority_queue<val_loc_t, std::vector<val_loc_t>, comp_val> queue;
00331         for (size_t i = 0; i < get_size(); i++) {
00332             T val = get(i);
00333             queue.push(val_loc_t(val, i));
